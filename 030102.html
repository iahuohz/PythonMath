案例2：偏差相关的计算¶

  • (deviation):, $ dev(x) = x_i - \bar{x}, i=1,2,,\cdots,n $
  • (variance):, $ var(x) = \dfrac{\sum_{i=1}^{n} (x^{(i)} - \bar{x})^2}{n} $
    • (unbiased estimator):n 1
    • (biased estimator):n
  • (standard variance):, $ std(x)=\sqrt{var(x)} $
  • (covariance):,, , $ cov(x, y)=\dfrac{\sum_i^n (\bar{x}-x^{(i)})(\bar{y}-y^{(i)})}{n} $
    • $ \bar{y} $y
    • $ \bar{x} $x
    • $ x^{(i)} $xi
    • $ y^{(i)} $yi
  • (correlation):(),,$ cor(x, y) = \dfrac{cov(x, y)}{std(x)*std(y)} $

arr = np.arange(1, 6, 1)

def deviation(array):
    """"""
    avg = np.mean(array)
    return [el - avg for el in array]

print("", deviation(arr))        # :[-2,-1,0,1,2]

def unbiased_variance(array):
    """"""
    deviations = deviation(array)
    return sum([el**2 for el in deviations]) / (len(array) - 1)

def biased_variance(array):
    """"""
    deviations = deviation(array)
    return sum([el**2 for el in deviations]) / len(array)

print("", unbiased_variance(arr))  # 2.5
print("(numpy)", np.var(arr, ddof=1))     # ddof=1
print("", biased_variance(arr))    # 2.0

def unbiased_covariance(arrayX, arrayY):
    """"""
    deX = deviation(arrayX)
    deY = deviation(arrayY)
    return sum([el1 * el2 for el1, el2 in zip(deX, deY)]) / (len(arrayX) - 1)

def unbiased_correlation(arrayX, arrayY):
    """"""
    deX = np.std(arrayX, ddof=1)
    deY = np.std(arrayY, ddof=1)
    if(deX > 0 and deY > 0):
        return unbiased_covariance(arrayX, arrayY) / (deX * deY)
    else:
        return 0

案例3：查看正态分布的概率密度函数和累积分布函数¶

  • $ PDF(x)=\dfrac{1}{\sigma\sqrt{2\pi}} e^{-\frac{(x-\mu)^2}{2\sigma^2}} $
  • $ CDF(x)=\dfrac{1}{\sigma\sqrt{2\pi}} \int_{-\infty}^{x} e^{-\frac{(x-\mu)^2}{2\sigma^2}}dx $

$ \sigma $ $ \mu $PDFCDF

import numpy as np
from scipy.integrate import quad
import matplotlib as mpl
import matplotlib.pyplot as plt
%matplotlib inline

# 
def normal_pdf(x, mu, sigma):
    return 1 / (sigma * np.sqrt(2 * np.pi)) * np.exp(-(x - mu)**2 / (2*sigma**2))

# (x)
def normal_integrate(x, mu, sigma):
    r, _ = quad(lambda t : np.exp(-(t - mu)**2 / (2*sigma**2)), -np.inf, x)
    return 1 / (sigma * np.sqrt(2 * np.pi)) * r

# x
def normal_cdf(x, mu, sigma):
    return [normal_integrate(i, mu, sigma) for i in x]

mpl.rcParams['font.sans-serif']=['SimHei']  #
mpl.rcParams['axes.unicode_minus']=False    #

x = np.linspace(-6, 6, 100)

# PDF
handles = []
labels = []
plt.subplot(1,2,1)
for mu in [0, -1]:
    for sigma in [0.5, 1, 2]:        
        y1 = normal_pdf(x, mu, sigma)
        f, = plt.plot(x, y1)
        handles.append(f)
        labels.append("mu={0}, sigma={1}".format(mu, sigma))
plt.title(u"PDF")
plt.legend(handles=handles, labels=labels)

# CDF
handles.clear()
labels.clear()
plt.subplot(1,2,2)
for mu in [0]:
    for sigma in [0.5, 1, 2]:        
        y1 = normal_cdf(x, mu, sigma)
        f, = plt.plot(x, y1)
        handles.append(f)
        labels.append("mu={0}, sigma={1}".format(mu, sigma))
plt.legend(handles=handles, labels=labels)
plt.title(u"CDF")

plt.show()

total_count = 100000
section_count = 20
section_max_value = 2.0 #-2.0~2.0
section_min_value = -2.0
section_interval = (section_max_value - section_min_value) / section_count
data = np.random.randn(total_count)
counts = np.array(section_count)
x = np.arange(section_min_value, section_max_value, section_interval)
y = np.zeros(len(x))
# -2.0~-1.9, -1.9~-1.8...... -0.1~0.0, 0.0~0.1, 0.1~0.2....1.9~2.0data
for value in data:
    if value > section_max_value or value < section_min_value:
        continue
    # 
    index = int((value - section_min_value) / section_interval) 
    y[index] += 1
    
plt.xticks(np.arange(section_min_value, section_max_value, section_interval))
bar_width = section_interval/2
plt.bar(x+bar_width/2, y, bar_width)
plt.xlabel("random data sctions")
plt.ylabel("random data counts")
